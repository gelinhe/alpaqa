65CasADiProblem<Conf>::CasADiProblem(const std::string &so_name, length_t n,
66 length_t m, length_t p, bool second_order)
67 : Problem<Conf>{n, m} {
68 using namespace casadi_loader;
69 auto load_g_unknown_dims =
70 [&]() -> std::optional<CasADiFunctionEvaluator<Conf, 2, 1>> {
71 casadi::Function gfun = casadi::external("g", so_name);
72 using namespace std::literals::string_literals;
73 if (gfun.n_in() != 2)
74 throw std::invalid_argument(
75 "Invalid number of input arguments: got "s +
76 std::to_string(gfun.n_in()) + ", should be 2.");
77 if (gfun.n_out() > 1)
78 throw std::invalid_argument(
79 "Invalid number of output arguments: got "s +
80 std::to_string(gfun.n_in()) + ", should be 0 or 1.");
81 if (gfun.size2_in(0) != 1)
82 throw std::invalid_argument(
83 "First input argument should be a column vector.");
84 if (gfun.size2_in(1) != 1)
85 throw std::invalid_argument(
86 "Second input argument should be a column vector.");
87 if (gfun.n_out() == 1 && gfun.size2_out(0) != 1)
88 throw std::invalid_argument(
89 "First output argument should be a column vector.");
90 if (n <= 0)
91 n = gfun.size1_in(0);
92 if (m <= 0 && gfun.n_out() == 1)
93 m = gfun.size1_out(0);
94 if (p <= 0)
95 p = gfun.size1_in(1);
96 if (gfun.n_out() == 0) {
97 if (m != 0)
98 throw std::invalid_argument(
99 "Function g has no outputs but m != 0");
100 return std::nullopt;
101 }
102 CasADiFunctionEvaluator<Conf, 2, 1> g{std::move(gfun)};
103 g.validate_dimensions({dim(n, 1), dim(p, 1)}, {dim(m, 1)});
104 return std::make_optional(std::move(g));
105 };
106
107 auto load_g_known_dims = [&] {
108 CasADiFunctionEvaluator<Conf, 2, 1> g{casadi::external("g", so_name),
109 {dim(n, 1), dim(p, 1)},
110 {dim(m, 1)}};
111 return g;
112 };
113
114 std::optional<CasADiFunctionEvaluator<Conf, 2, 1>> g =
115 (n <= 0 || m <= 0 || p <= 0)
116 // If not all dimensions are specified, load the function "g" to
117 // determine the missing dimensions.
118 ? wrap_load(so_name, "g", load_g_unknown_dims)
119 // Otherwise, load the function "g" and compare its dimensions to
120 // the dimensions specified by the user.
121 : wrap_load(so_name, "g", load_g_known_dims);
122
123 this->n = n;
124 this->m = m;
125 this->param = vec::Constant(p, alpaqa::NaN<Conf>);
126
127 impl = std::make_unique<CasADiFunctionsWithParam<Conf>>(
128 CasADiFunctionsWithParam<Conf>{
129 wrapped_load<CasADiFunctionEvaluator<Conf, 2, 1>>( //
130 so_name, "f", dims(n, p), dims(1)),
131 // wrapped_load<CasADiFunctionEvaluator<6, 1>>( //
132 // so_name, "grad_psi", dims(n, p, m, m, m, m), dims(n)),
133 wrapped_load<CasADiFunctionEvaluator<Conf, 6, 2>>( //
134 so_name, "psi_grad_psi", dims(n, p, m, m, m, m), dims(1, n)),
135 std::nullopt,
136 std::nullopt,
137 });
138
139 if (g)
140 impl->constr = std::make_optional(
141 typename CasADiFunctionsWithParam<Conf>::ConstrFun{
142 std::move(*g),
143 wrapped_load<CasADiFunctionEvaluator<Conf, 3, 1>>( //
144 so_name, "grad_L", dims(n, p, m), dims(n)),
145 wrapped_load<CasADiFunctionEvaluator<Conf, 6, 2>>( //
146 so_name, "psi", dims(n, p, m, m, m, m), dims(1, m)),
147 });
148
149 if (second_order)
150 impl->hess =
151 std::make_optional(typename CasADiFunctionsWithParam<Conf>::HessFun{
152 wrapped_load<CasADiFunctionEvaluator<Conf, 4, 1>>( //
153 so_name, "hess_L_prod", dims(n, p, m, n), dims(n)),
154 wrapped_load<CasADiFunctionEvaluator<Conf, 3, 1>>( //
155 so_name, "hess_L", dims(n, p, m), dims(dim(n, n))),
156 });
157}
