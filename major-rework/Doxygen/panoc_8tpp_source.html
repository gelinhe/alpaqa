297 }

301 direction_provider.changed_γ(γₖ₊₁, γₖ);

303 s.lbfgs_rejected += not direction_provider.update(xₖ, xₖ₊₁, pₖ, pₖ₊₁,
304 grad_ψₖ₊₁, C, γₖ₊₁);

308 no_progress = xₖ == xₖ₊₁ ? no_progress + 1 : 0;
309
310 // Advance step --------------------------------------------------------
311 Lₖ = Lₖ₊₁;
312 γₖ = γₖ₊₁;
313
314 ψₖ = ψₖ₊₁;
315 ψx̂ₖ = ψx̂ₖ₊₁;
316 φₖ = φₖ₊₁;
317
318 xₖ.swap(xₖ₊₁);
319 x̂ₖ.swap(x̂ₖ₊₁);
320 ŷx̂ₖ.swap(ŷx̂ₖ₊₁);
321 pₖ.swap(pₖ₊₁);
322 grad_ψₖ.swap(grad_ψₖ₊₁);
