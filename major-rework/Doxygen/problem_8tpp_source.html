140template <Config Conf>
141auto ProblemBase<Conf>::calc_ŷ_dᵀŷ(rvec g_ŷ, crvec y, crvec Σ) const -> real_t {
142 if (Σ.size() == 1) {
143 // ζ = g(x) + Σ⁻¹y
144 g_ŷ += (1 / Σ(0)) * y;
145 // d = ζ - Π(ζ, D)
146 g_ŷ = projecting_difference(g_ŷ, get_D());
147 // dᵀŷ, ŷ = Σ d
148 real_t dᵀŷ = Σ(0) * g_ŷ.dot(g_ŷ);
149 g_ŷ *= Σ(0);
150 return dᵀŷ;
151 } else {
152 // ζ = g(x) + Σ⁻¹y
153 g_ŷ += Σ.asDiagonal().inverse() * y;
154 // d = ζ - Π(ζ, D)
155 g_ŷ = projecting_difference(g_ŷ, get_D());
156 // dᵀŷ, ŷ = Σ d
157 real_t dᵀŷ = 0;
158 for (unsigned i = 0; i < m; ++i) {
159 dᵀŷ += g_ŷ(i) * Σ(i) * g_ŷ(i); // TODO: vectorize
160 g_ŷ(i) = Σ(i) * g_ŷ(i);
161 }
162 return dᵀŷ;
163 }
