203/**
204 * @brief Problem description for minimization problems.
205 *
206 * @f[ \begin{aligned}
207 * & \underset{x}{\text{minimize}}
208 * & & f(x) &&&& f : \R^n \rightarrow \R \\
209 * & \text{subject to}
210 * & & \underline{x} \le x \le \overline{x} \\
211 * &&& \underline{z} \le g(x) \le \overline{z} &&&& g :
212 * \R^n \rightarrow \R^m
213 * \end{aligned} @f]
214 * <br>
215 * @f[ \begin{aligned}
216 * C &\defeq \defset{x\in\R^n}{\underline x \le x \le \overline x} \\
217 * D &\defeq \defset{z\in\R^m}{\underline z \le z \le \overline z} \\
218 * \end{aligned} @f]
219 *
220 * The functions in the “Basic functions” section have to be implemented by the
221 * user. Functions in the “Combined evaluations” and “Augmented Lagrangian”
222 * sections are optional, by default, they are computed by evaluating the
223 * “Basic functions”, but performance can be improved by providing functions
224 * that directly compute multiple quantities at once.
225 */
226template <Config Conf = DefaultConfig>
227class Problem : public ProblemBase<Conf> {
228 public:
229 USING_ALPAQA_CONFIG(Conf);
230 using ProblemBase = alpaqa::ProblemBase<Conf>;
231 using Box = typename ProblemBase::Box;
232 using ProblemBase::m;
233 using ProblemBase::n;
234
235 /// The parameter vector
236 vec param;
237 /// Constraints of the decision variables, @f$ x \in C @f$
238 Box C{vec::Constant(this->n, +inf<Conf>),
239 vec::Constant(this->n, -inf<Conf>)};
240 /// Other constraints, @f$ g(x) \in D @f$
241 Box D{vec::Constant(this->m, +inf<Conf>),
242 vec::Constant(this->m, -inf<Conf>)};
243
244 Problem(length_t n, ///< Number of decision variables
245 length_t m, ///< Number of constraints
246 vec param) ///< Problem parameter
247 : ProblemBase{n, m}, param{std::move(param)} {}

252 : Problem{n, m, vec::Constant(p, NaN<config_t>)} {}

254 Problem(length_t n, ///< Number of decision variables
