11template <Config Conf>
12bool LBFGS<Conf>::update_valid(const Params &params, real_t yᵀs, real_t sᵀs,
13 real_t pᵀp) {
14 // Check if this L-BFGS update is accepted
15 if (sᵀs <= params.min_abs_s)
16 return false;
17 if (not std::isfinite(yᵀs))
18 return false;
19 real_t a_yᵀs = params.force_pos_def ? yᵀs : std::abs(yᵀs);
20 if (a_yᵀs <= params.min_div_fac * sᵀs)
21 return false;
22
23 // CBFGS condition: https://epubs.siam.org/doi/10.1137/S1052623499354242
24 if (params.cbfgs) {
25 const real_t α = params.cbfgs.α;
26 const real_t ϵ = params.cbfgs.ϵ;
27 // Condition: yᵀs / sᵀs >= ϵ ‖p‖^α
28 bool cbfgs_cond = a_yᵀs >= sᵀs * ϵ * std::pow(pᵀp, α / 2);
29 if (not cbfgs_cond)
30 return false;
31 }
32
33 return true;
34}

36template <Config Conf>
37bool LBFGS<Conf>::update_sy_impl(const auto &s, const auto &y, real_t pₖ₊₁ᵀpₖ₊₁,
38 bool forced) {
39 real_t yᵀs = y.dot(s);
40 real_t ρ = 1 / yᵀs;
41 if (not forced) {
42 real_t sᵀs = s.squaredNorm();
43 if (not update_valid(params, yᵀs, sᵀs, pₖ₊₁ᵀpₖ₊₁))
44 return false;
45 }
46
47 // Store the new s and y vectors
48 sto.s(idx) = s;
49 sto.y(idx) = y;
50 sto.ρ(idx) = ρ;
51
52 // Increment the index in the circular buffer
53 idx = succ(idx);
54 full |= idx == 0;
55
56 return true;
57}

73template <Config Conf>
74bool LBFGS<Conf>::apply(rvec q, real_t γ) const {
75 // Only apply if we have previous vectors s and y
76 if (idx == 0 && not full)
77 return false;
78
79 // If the step size is negative, compute it as sᵀy/yᵀy
80 if (γ < 0) {
81 auto new_idx = pred(idx);
82 real_t yᵀy = y(new_idx).squaredNorm();
83 γ = 1 / (ρ(new_idx) * yᵀy);
84 }
85
86 foreach_rev([&](index_t i) {
87 α(i) = ρ(i) * s(i).dot(q);
88 q -= α(i) * y(i);
89 });
90
91 // r ← H₀ q
92 q *= γ;
93
94 foreach_fwd([&](index_t i) {
95 real_t β = ρ(i) * y(i).dot(q);
96 q -= (β - α(i)) * s(i);
97 });
98
99 return true;
100}

102template <Config Conf>
103bool LBFGS<Conf>::apply_masked_impl(rvec q, real_t γ, const auto &J) {
104 // Only apply if we have previous vectors s and y
105 if (idx == 0 && not full)
106 return false;
107 const bool fullJ = q.size() == static_cast<index_t>(J.size());
108
109 if (params.cbfgs)
110 throw std::invalid_argument("CBFGS check not supported when using "
111 "masked version of LBFGS::apply_masked()");
112
113 // Eigen 3.3.9 doesn't yet support indexing using a vector of indices
114 // so we'll have to do it manually.
115 // TODO: Abstract this away in an expression template / nullary expression?
116 // Or wait for Eigen update?
117 // Update: Eigen 3.4's indexing seems significantly slower, so the manual
118 // for loops stay for now.
119
120 // Dot product of two vectors, adding only the indices in set J
121 const auto dotJ = [&J, fullJ](const auto &a, const auto &b) {
122 if (fullJ) {
123 return a.dot(b);
124 } else {
125 real_t acc = 0;
126 for (auto j : J)
127 acc += a(j) * b(j);
128 return acc;
129 }
130 };
131 // y -= a x, scaling and subtracting only the indices in set J
132 const auto axmyJ = [&J, fullJ](real_t a, const auto &x, auto &y) {
133 if (fullJ) {
134 y -= a * x;
135 } else {
136 for (auto j : J)
137 y(j) -= a * x(j);
138 }
139 };
140 // x *= a, scaling only the indices in set J
141 const auto scalJ = [&J, fullJ](real_t a, auto &x) {
142 if (fullJ) {
143 x *= a;
144 } else {
145 for (auto j : J)
146 x(j) *= a;
147 }
148 };
149
150 foreach_rev([&](index_t i) {
151 // Recompute ρ, it depends on the index set J. Note that even if ρ was
152 // positive for the full vectors s and y, that's not necessarily the
153 // case for the smaller vectors s(J) and y(J).
154 real_t yᵀs = dotJ(s(i), y(i));
155 real_t sᵀs = dotJ(s(i), s(i));
156 ρ(i) = 1 / yᵀs;
157 // Check if we should include this pair of vectors
158 if (not update_valid(params, yᵀs, sᵀs, 0)) {
159 ρ(i) = NaN<config_t>;
160 return;
161 }
162
163 α(i) = ρ(i) * dotJ(s(i), q); // αᵢ = ρᵢ〈sᵢ, q〉
164 axmyJ(α(i), y(i), q); // q -= αᵢ yᵢ
165
166 if (γ < 0) {
167 // Compute step size based on most recent valid yᵀs/yᵀy
168 real_t yᵀy = dotJ(y(i), y(i));
169 γ = 1 / (ρ(i) * yᵀy);
170 }
171 });
172
173 // If all ρ == 0, fail
174 if (γ < 0)
175 return false;
176
177 // r ← H₀ q
178 scalJ(γ, q); // q *= γ
179
180 foreach_fwd([&](index_t i) {
181 if (std::isnan(ρ(i)))
182 return;
183 real_t β = ρ(i) * dotJ(y(i), q); // βᵢ = ρᵢ〈yᵢ, q〉
184 axmyJ(β - α(i), s(i), q); // q -= (βᵢ - αᵢ) sᵢ
185 });
186
187 return true;
188}
