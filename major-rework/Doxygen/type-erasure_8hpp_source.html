17/// Struct that stores the size of a polymorphic object, as well as pointers to
18/// functions to copy, move or destroy the object.
19/// Inherit from this struct to add useful functions.
20struct BasicVTable {
21 size_t size = static_cast<size_t>(0xDEADBEEFDEADBEEF);
22 void (*copy)(const void *self, void *storage) = nullptr;
23 void (*move)(void *self, void *storage) = nullptr;
24 void (*destroy)(void *self) = nullptr;
25
26 BasicVTable() = default;
27
28 template <class T>
29 BasicVTable(VTableTypeTag<T>) {
30 size = sizeof(T);
31 copy = [](const void *self, void *storage) {
32 new (storage) T(*std::launder(reinterpret_cast<const T *>(self)));
33 };
34 move = [](void *self, void *storage) {
35 new (storage)
36 T(std::move(*std::launder(reinterpret_cast<T *>(self))));
37 };
38 destroy = [](void *self) {
39 std::launder(reinterpret_cast<T *>(self))->~T();
40 };
41 }
42};

193 /// Ensure that storage is available, either by using the small buffer if
194 /// it is large enough, or by calling the allocator.
195 void ensure_storage() {

197 self = vtable.size <= small_buffer_size

214 template <bool CopyAllocator>
215 void do_copy_assign(const TypeErased &other) {
216 constexpr bool prop_alloc =
217 allocator_traits::propagate_on_container_copy_assignment::value;
218 if constexpr (CopyAllocator && prop_alloc)
219 allocator = other.allocator;
220 vtable = other.vtable;
221 ensure_storage();
222 vtable.copy(other.self, self);
223 }

225 template <bool MoveAllocator>
226 void do_move_assign(TypeErased &&other) {
227 constexpr bool prop_alloc =
228 allocator_traits::propagate_on_container_move_assignment::value;
229 if constexpr (MoveAllocator && prop_alloc)
230 allocator = std::move(other.allocator);
231 vtable = other.vtable;
232 // If dynamically allocated, simply steal storage
233 if (vtable.size > small_buffer_size) {
234 // If we can steal the storage because of equal allocators, do so
235 if (allocator == other.allocator) {
236 self = std::exchange(other.self, nullptr);
237 }
238 // If the allocators are not the same, we cannot steal the storage,
239 // so do an explicit move
240 else {
241 self = allocator.allocate(vtable.size);
242 vtable.move(other.self, self);
243 other.cleanup();
244 }
245 }
246 // Otherwise, use the small buffer and do an explicit move
247 else if (other.self) {
248 self = small_buffer.data();
249 vtable.move(other.self, self);
250 other.cleanup();
251 }
252 }

258 vtable = VTable{VTableTypeTag<T>{}};

261 }
